class NumberThread extends Thread {
    public void run() {
        for (int i = 1; i <= 10; i++) {
            System.out.println("Number: " + i);
        }
    }

    public static void main(String[] args) {
        NumberThread t1 = new NumberThread();
        t1.start();
    }
}
Output:
Number: 1
Number: 2
Number: 3
...
Number: 10
